14use ::core::fmt;
15
16pub trait FromPrimitive: Sized {
17    type Primitive: Copy + Eq;
18
19    fn from_primitive(number: Self::Primitive) -> Self;
20}
21
22pub trait TryFromPrimitive: Sized {
23    type Primitive: Copy + Eq + fmt::Debug;
24
25    const NAME: &'static str;
26
27    fn try_from_primitive(number: Self::Primitive) -> Result<Self, TryFromPrimitiveError<Self>>;
28}
29
30pub struct TryFromPrimitiveError<Enum: TryFromPrimitive> {
31    pub number: Enum::Primitive,
32}
33
34impl<Enum: TryFromPrimitive> Copy for TryFromPrimitiveError<Enum> {}
35impl<Enum: TryFromPrimitive> Clone for TryFromPrimitiveError<Enum> {
36    fn clone(&self) -> Self {
37        TryFromPrimitiveError {
38            number: self.number,
39        }
40    }
41}
42impl<Enum: TryFromPrimitive> Eq for TryFromPrimitiveError<Enum> {}
43impl<Enum: TryFromPrimitive> PartialEq for TryFromPrimitiveError<Enum> {
44    fn eq(&self, other: &Self) -> bool {
45        self.number == other.number
46    }
47}
48impl<Enum: TryFromPrimitive> fmt::Debug for TryFromPrimitiveError<Enum> {
49    fn fmt(&self, fmt: &mut fmt::Formatter) -> fmt::Result {
50        fmt.debug_struct("TryFromPrimitiveError")
51            .field("number", &self.number)
52            .finish()
53    }
54}
55impl<Enum: TryFromPrimitive> fmt::Display for TryFromPrimitiveError<Enum> {
56    fn fmt(&self, stream: &'_ mut fmt::Formatter<'_>) -> fmt::Result {
57        write!(
58            stream,
59            "No discriminant in enum `{name}` matches the value `{input:?}`",
60            name = Enum::NAME,
61            input = self.number,
62        )
63    }
64}
