9pub fn compile(source: &str) {
10    let mut scanner = Scanner::new(source);
11    let mut line = -1;
12
13    loop {
14        let token = scanner.scan_token();
15
16        if token.line != line {
17            print!("{:4} ", token.line);
18            line = token.line;
19        } else {
20            print!("   | ");
21        }
22
23        println!("{:?} '{}'", token.token_type, token.lexeme);
24
25        if token.token_type == TokenType::EOF {
26            break;
27        }
28    }
29}

60/// Errors that can occur during compilation.
61#[derive(Error, Clone, Debug)]
62pub enum CompileError {
63    #[error("Too many constants")]
64    TooManyConstants,
65}
66
67/// The maximum number of constants that can be stored in a chunk.
68const MAX_CONSTANTS: usize = 0xFFFFFF; // 24 bits
69
70impl Chunk {
71    pub fn new() -> Self {
72        Self {
73            code: Vec::new(),
74            lines: Vec::new(),
75            constants: Vec::new(),
76        }
77    }
78
79    pub fn read(&self, idx: usize) -> u8 {
80        self.code[idx]
81    }
82
83    pub fn write<T>(&mut self, data: T, line: usize)
84    where
85        T: Into<u8>,
86    {
87        self.code.push(data.into());
88        self.lines.push(line);
89    }
90
91    pub fn write_constant<T>(&mut self, value: T, line: usize) -> Result<(), CompileError>
92    where
93        T: Into<Value>,
94    {
95        let index = self.add_constant(value.into())?;
96        let opcode = if index > u8::MAX as usize {
97            OpCode::ConstantLong
98        } else {
99            OpCode::Constant
100        };
101
102        self.write(opcode, line);
103
104        if opcode == OpCode::Constant {
105            self.write(index as u8, line);
106        } else {
107            let low = (index & 0xFF) as u8;
108            let mid = ((index >> 8) & 0xFF) as u8;
109            let high = ((index >> 16) & 0xFF) as u8;
110            self.write(high, line);
111            self.write(mid, line);
112            self.write(low, line);
113        }
114
115        Ok(())
116    }
117
118    fn add_constant(&mut self, value: Value) -> Result<usize, CompileError> {
119        if self.constants.len() >= MAX_CONSTANTS {
120            return Err(CompileError::TooManyConstants);
121        }
122
123        self.constants.push(value);
124        Ok(self.constants.len() - 1)
125    }
126
127    /// Disassembles the chunk, printing the given header to identify it.
128    pub fn disassemble(&self, name: &str) {
129        eprintln!("== {} ==", name);
130
131        let mut offset = 0;
132        while offset < self.code.len() {
133            offset = self.disassemble_instruction(offset);
134        }
135    }
136
137    /// Disassemble the instruction at the given offset.
138    pub fn disassemble_instruction(&self, offset: usize) -> usize {
139        eprint!("{:04} ", offset);
140        let line = self.lines[offset];
141
142        if offset > 0 && line == self.lines[offset - 1] {
143            eprint!("   | ");
144        } else {
145            eprint!("{:4} ", line);
146        }
147
148        let instruction = self.code[offset];
149        let opcode = OpCode::try_from(instruction);
150        match opcode {
151            Ok(OpCode::Constant) => constant_instruction("OP_CONSTANT", self, offset),
152            Ok(OpCode::ConstantLong) => constant_long_instruction("OP_CONSTANT_LONG", self, offset),
153
154            Ok(OpCode::Add) => simple_instruction("OP_ADD", offset),
155            Ok(OpCode::Subtract) => simple_instruction("OP_SUBTRACT", offset),
156            Ok(OpCode::Multiply) => simple_instruction("OP_MULTIPLY", offset),
157            Ok(OpCode::Divide) => simple_instruction("OP_DIVIDE", offset),
158            Ok(OpCode::Negate) => simple_instruction("OP_NEGATE", offset),
159
160            Ok(OpCode::Return) => simple_instruction("OP_RETURN", offset),
161
162            Err(_) => {
163                eprintln!("Unknown opcode {}", instruction);
164                offset + 1
165            }
166        }
167    }
168}

170/// Disassembles a simple single-byte instruction.
171fn simple_instruction(name: &str, offset: usize) -> usize {
172    eprintln!("{}", name);
173
174    offset + 1
175}

177/// Disassembles a simple constant instruction.
178fn constant_instruction(name: &str, chunk: &Chunk, offset: usize) -> usize {
179    let constant = chunk.code[offset + 1];
180    let value = chunk.constants[constant as usize];
181    eprintln!("{:16} {:4} '{}'", name, constant, value);
182
183    offset + 2
184}
185
186/// Disassembles a more complex "long constant" instruction.
187fn constant_long_instruction(name: &str, chunk: &Chunk, offset: usize) -> usize {
188    let high = chunk.code[offset + 1] as usize;
189    let mid = chunk.code[offset + 2] as usize;
190    let low = chunk.code[offset + 3] as usize;
191    let constant = high << 16 | mid << 8 | low;
192    let value = chunk.constants[constant];
193    eprintln!("{:16} {:8} '{}'", name, constant, value);
194
195    offset + 4
196}
